Initialisation code

$(':checkbox[name="rendering_filter"]').click(function() {
    var filter = '',
        regexFilter = true,
        smartFilter = false;

    filter = $('[name="' + this.name + '"]:checked').map(function () {
        return this.value;
    }).toArray().join('|');

    $('#example').dataTable().fnFilter(filter, 0, regexFilter, smartFilter);
});
